# 后端实现示例: 代码解析器

import ast
from pathlib import Path
from typing import List, Dict, Any
from dataclasses import dataclass
from watchfiles import awatch
import libcst as cst

@dataclass
class GraphNode:
    """解析出的图节点"""
    id: str
    name: str
    func_name: str
    line: int

@dataclass
class GraphEdge:
    """解析出的图边"""
    source: str
    target: str
    condition: str | None = None

class LangGraphParser:
    """LangGraph 代码解析器"""

    def parse_file(self, file_path: str) -> Dict[str, Any]:
        """解析 Python 文件，提取 StateGraph 结构"""
        with open(file_path) as f:
            source = f.read()

        tree = ast.parse(source)
        nodes, edges = [], []

        for node in ast.walk(tree):
            if isinstance(node, ast.Call):
                # 解析 graph.add_node("name", func)
                if self._is_add_node(node):
                    nodes.append(self._parse_add_node(node))
                # 解析 graph.add_edge("from", "to")
                elif self._is_add_edge(node):
                    edges.append(self._parse_add_edge(node))
                # 解析 graph.add_conditional_edges(...)
                elif self._is_conditional_edge(node):
                    edges.extend(self._parse_conditional_edges(node))

        return self._to_react_flow(nodes, edges)

    def _to_react_flow(self, nodes: List[GraphNode],
                       edges: List[GraphEdge]) -> Dict[str, Any]:
        """转换为 React Flow 格式"""
        return {
            "nodes": [
                {
                    "id": n.id,
                    "type": "custom",
                    "position": {"x": i * 200, "y": 100},
                    "data": {"label": n.name, "func": n.func_name}
                }
                for i, n in enumerate(nodes)
            ],
            "edges": [
                {
                    "id": f"{e.source}-{e.target}",
                    "source": e.source,
                    "target": e.target,
                    "label": e.condition
                }
                for e in edges
            ]
        }


class LangGraphCodeGen:
    """LangGraph 代码生成器 (使用 LibCST 保留格式)"""

    def add_node(self, file_path: str, node_name: str, func_name: str):

### 10.2 关键决策理由

| 决策 | 选择 | 理由 |
|------|------|------|
| 前端框架 | React + Vite (非 Next.js) | 不需要 SEO，纯 SPA 足够 |
| Agent 框架 | LangGraph (定制) | 检查点、中断机制与我们设计一致 |
| LLM 接入 | LiteLLM | 统一接口，支持 100+ 模型 |
| 向量数据库 | Qdrant | 性能好、易部署、免费 |
| 部署方案 | Docker Compose | 简单够用，后续可迁移 K8s |
| 类型检查 | Pyright + Pydantic | 静态+运行时双重验证，接近 TS 体验 |
| LSP 集成 | Pyright LSP | 实时诊断，与类型检查统一 |
| 代码生成质量 | 多层验证流水线 | 语法→类型→lint→架构→沙箱执行 |

> 代码生成质量保障见 [backend/docs/CODE_STANDARDS.md](../backend/docs/CODE_STANDARDS.md) 与 [frontend/docs/FRONTEND_CODE_CHECK.md](../frontend/docs/FRONTEND_CODE_CHECK.md)。
